package model;

import lombok.Getter;
import lombok.Setter;

import java.util.ArrayList;
import java.util.List;

@Getter
@Setter
public class PizzaVendingMachine {
    private MachineState state = MachineState.IDLE;
    private List<Pizza> inventory = new ArrayList<>();
    private boolean pizzaDispensed;

    public void displayMenu() {
        System.out.println("displaying menu.");
    }

    public Pizza selectPizza(PizzaType type, PizzaSize size, CrustType crustType) {
        return new Pizza(type, size, crustType);
    }

    public void addToCart(Cart cart, Pizza pizza) {
        cart.addToCart(pizza);
    }

    public double calculateTotal(Cart cart) {
        return cart.calculateTotal();
    }

    public boolean processPayment(Payment payment) {
        return payment.processPayment();
    }

    public void dispensePizza() {
        System.out.println("dispensing pizza.");
        pizzaDispensed = true;
    }

    public void performMaintenance() {
        state = MachineState.MAINTENANCE;
        System.out.println("performing maintenance.");
    }
}

package model;

public class Application {
    public static void main(String... args) {
        PizzaVendingMachine machine = new PizzaVendingMachine();

        machine.getInventory().add(new Pizza(PizzaType.PEPPERONI, PizzaSize.MEDIUM, CrustType.THIN));
        machine.getInventory().add(new Pizza(PizzaType.MARGHERITA, PizzaSize.LARGE, CrustType.THICK));

        machine.displayMenu();

        Pizza selectedPizza = machine.selectPizza(PizzaType.PEPPERONI, PizzaSize.MEDIUM, CrustType.THIN);

        Cart cart = new Cart();
        machine.addToCart(cart, selectedPizza);

        double total = machine.calculateTotal(cart);
        System.out.println("total amount: " + total);

        Payment payment = new Payment(PaymentMethod.CREDIT_CARD, total);
        if (machine.processPayment(payment)) {
            machine.dispensePizza();
        } else {
            System.out.println("payment failed.");
        }
    }
}
