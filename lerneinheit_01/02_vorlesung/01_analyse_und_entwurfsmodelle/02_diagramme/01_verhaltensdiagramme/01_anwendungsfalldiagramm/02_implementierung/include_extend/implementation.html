import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;

@Slf4j
@RequiredArgsConstructor
public class PlaceOrder extends UseCase {
    private final SelectPizza selectPizza;
    private final AddToppings addToppings;
    private final VerifyAvailability verifyAvailability;

    public void execute() {
        log.info("placing order.");
        selectPizza.execute();                           // <<include>>` relationship
        addToppings.execute();                           // <<include>>` relationship
        verifyAvailability.execute();                    // <<include>>` relationship
        log.info("order placed successfully.");
    }
}

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;

@Slf4j
@RequiredArgsConstructor
public class ProcessPayment extends UseCase {
    private final ApplyDiscount applyDiscount;
    private final HandlePaymentFailure handlePaymentFailure;

    public void execute() {
        log.info("processing payment.");
        boolean isPaymentSuccessful = true;

        if (applyDiscount.isDiscountApplicable()) {       // <<extend>> relationship
            applyDiscount.execute();
        }

        if (!isPaymentSuccessful) {                       // <<extend>> relationship
            handlePaymentFailure.execute();
        } else {
            log.info("payment processed successfully.");
        }
    }
}

import lombok.Data;
import lombok.EqualsAndHashCode;
import lombok.extern.slf4j.Slf4j;

@Slf4j
@Data
@EqualsAndHashCode(callSuper = true)
// <<extend>> relationship
public class ApplyDiscount extends UseCase {
    private final boolean isDiscountApplicable;

    public void execute() {
        if (isDiscountApplicable) {
            log.info("applying discount.");
        }
    }
}

public class Application {
    public static void main(String... args) {
        SelectPizza selectPizza = new SelectPizza();
        AddToppings addToppings = new AddToppings();
        VerifyAvailability verifyAvailability = new VerifyAvailability();

        PlaceOrder placeOrder = new PlaceOrder(selectPizza, addToppings, verifyAvailability);

        ApplyDiscount applyDiscount = new ApplyDiscount(true);
        HandlePaymentFailure handlePaymentFailure = new HandlePaymentFailure();

        ProcessPayment processPayment = new ProcessPayment(applyDiscount, handlePaymentFailure);

        placeOrder.execute();
        processPayment.execute();
    }
}
