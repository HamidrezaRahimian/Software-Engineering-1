package good;

import java.util.ArrayList;
import java.util.List;

public class Application {
    public static void playVideoInAllMediaPlayers(List<VideoMediaPlayer> videoMediaPlayerList) {
        for (VideoMediaPlayer player : videoMediaPlayerList) {
            player.playVideo();
        }
    }

    public static void main(String... args) {
        List<VideoMediaPlayer> allPlayers = new ArrayList<>();
        allPlayers.add(new VLCMediaPlayer());
        allPlayers.add(new DVIMediaPlayer());
        playVideoInAllMediaPlayers(allPlayers);
    }
}
