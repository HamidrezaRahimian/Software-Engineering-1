package good;

// high-level module
public class Smartphone {
    private final AIProcessor aiProcessor;

    public Smartphone(AIProcessor aiProcessor) {
        this.aiProcessor = aiProcessor;  // dependency injection of the abstraction
    }

    public void useAI() {
        aiProcessor.processAI();
    }
}
